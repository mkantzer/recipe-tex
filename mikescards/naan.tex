\begin{recipe}{Naan}{15 Min bake, 30min/2hr prep}{King Arthur Baking}

\begin{parcolumns}[colwidths={1=80pt}, rulebetween]{2}
  \colchunk{
    \begin{enumerate}
      \item \textbf{DOUGH:}
      \item 180g AP Flour
      \item 90g water, lukewarm
      \item 142g warm water
      \item 71g Greek Yogurt
      \item 28g ghee/butter, melted
      \item 1\frac{1}{2}tsp instant yeast
      \item 1 tsp granulated sugar
      \item 1 tsp salt
      \item \textbf{TOPPING:}
      \item 28g ghee/butter, melted
      \item 2 tsp nigella seeds (opt)
      \item 1/4 cup cilantro, chopped (opt)
      \item \textbf{FOR GARLIC NAAN:}
      \item 5-6 cloves garlic, minced
    \end{enumerate}
  }
  \colchunk{
    \begin{enumerate}
      \item Combine all of the \textbf{DOUGH} ingredients in a large bowl.
      \item Mix till shaggy
      \item Knead until smooth, bouncy, slightly tacky
      \item Place in greased bowl, cover, let rise till doubled. 50-60 min
      \item Divide dough into 8 pieces. Shape into balls, cover, rest for 20 min
      \item Prehead cast iron, med-high
      \item Floured surface, stretch/roll balls to:  THICK: 6" round / 7" oblong , THIN: 8" round, 10" oblong.
      \item \textbf{For GARLIC NAAN:} sprinkle garlic on circles, press in firmly
      \item Cook naan on griddle 30-40 seconds until top bubbles and bottom browns. Flip, brown other side 40-50 sec. Check regularly
      \item Remove from pan, brush w/ melted ghee/butter. Sprinkle w/ seeds/cilantro
      \item Keep finished naan in 200 \textdegree F oven or wrapped in foil till served
      \item Store well wrapped at room temp for 2 days. Freeze for longer
    \end{enumerate}
  }
\end{parcolumns}



\end{recipe}
