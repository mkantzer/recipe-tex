\begin{recipe}{Monkey Bread}{3.5 hours}{Sally'sBakingAddiction}

\begin{parcolumns}[colwidths={1=100pt}, rulebetween]{2}
  \colchunk{
    \begin{enumerate}
      \item 1.5 Cup Whole Milk, 110 \textdegree F
      \item 2.25 tsp Yeast
      \item 50g granulated sugar
      \item 2 large egg
      \item 5 TBSP unsalted butter, melted
      \item 1 tsp Salt
      \item 625 g AP Flour
      \item \textbf{coating}
      \item 12 TBSP unsalted butter, divided
      \item 250g granulated sugar
      \item 1 TBSP cinnamon
      \item 130 g brown sugar
      \item 1 tsp vanilla
      \item \textbf{icing}
      \item 120g confectioners sugar
      \item 3 TBSP whole Milk
      \item 0.5 tsp vanilla extract
    \end{enumerate}
  }
  \colchunk{
    \begin{enumerate}
      \item Whisk milk, yeast, sugar in bowl w/ paddle. Cover, sit 5 min.
      \item Add eggs, butter, salt, 1 cup flour. low for 30. scrape, add remaining flour.
      \item Beat medium, till together, 2 min.
      \item Beat for added 2 min to knead dough
      \item 1st rise: grease bowl, place in dough. Cover. Allow rise, 1-2 hour.
      \item Grease 10-12 bundt pan.
      \item Melt 1 stick butter. In another bowl, mix granulated sugar and cinnamon. 
      \item Shape dough: punch down, roll small balls, 40-45 total. Dip each in butter and roll in sugar.
      \item Place in bundt pan as you go.
      \item Cover pan, resting 20 minutes. 
      \item Oven rack, low position, 350/textdegree F , preheat
      \item Melt 0.25 cup butter, whisk in brown sugar, vanilla. POour over bread. 
      \item Bake, 35-45 min. Cover with foil if top browns too fast. Cool, 5-10 min. Invert.
      \item Icing: whisk all, then drizzle over. 
    \end{enumerate}
  }
\end{parcolumns}

\end{recipe}
