\begin{recipe}{Sourdough}{24ish h}{https://makefastworkshop.com/hacks/?p=20200515}
\begin{parcolumns}[colwidths={1=70pt}, rulebetween]{2}
  \colchunk{
    \begin{enumerate}
      \item \textbf{Settings}
      \item Full Sourdough
      \item 2x 900g Boule 
      \item Strong AP Flour
      \item No Secondary
      \item Medium Hydration
      \item 2\% salt
      \item Toothsome
      \item Dutch Oven
      \item 
      \item \textbf{Ingredients}
      \item 16g active starter
      \item 706g water, warm
      \item 1057g Sir Galahad flour
      \item 21g Kosher Salt
    \end{enumerate}
  }
  \colchunk{
    \begin{enumerate}
      \item \textbf{Dough}
      \item Combine ingredients (excpet salt) until shaggy \textbf{2-3m}.
      \item Cover (damp cloth) and autolyse: \textbf{15m}.
      \item Add salt and work in. Cover. 
      \item Stretch and fold ev \textbf{10m} w/ wet hands \textbf{4x}.
      \item 
      \item \textbf{Bulk Rise}
      \item Cover in warm spot, until 2x size. \textbf{16-20h}, depending. 
      \item 
      \item \textbf{Cut and Shape}
      \item Preheat oven w/ dutch oven to \textbf{500\textdegree F}.
      \item LIGHTLY flour half work surface (cutting). Leave other unfloured (shaping).
      \item Turn out dough. Cut in half. Form into balls. Rest \textbf{15m}.
      \item 
      \item \textbf{Second Rise}
      \item Shape into boule. Rest 30m or \textbf{2-4 hours}.
      \item 
      \item \textbf{Score and Bake}
      \item Center rack, lid on, \textbf{430\textdegree F}, \textbf{44m}.
      \item Remove lid, bake \textbf{8m}. Internal temp \textbf{205\textdegree F}
      \item Cool on wire rack \textbf{1h}
    \end{enumerate}
  }
\end{parcolumns}
\end{recipe}
