\begin{recipe}{Raspberry-Almond Rugelach}{3hr bake. 45 min Prep}{Serious Eats}

\begin{parcolumns}[colwidths={1=100pt}, rulebetween]{2}
  \colchunk{
    \begin{enumerate}
      \item \textbf{DOUGH:}
      \item 4.25oz AP Flour
      \item 4oz Butter, cubed
      \item 4oz Cream Cheese, cubed
      \item $\frac{1}{2}$ tsp Orange Zest
      \item $\frac{1}{8}$ Kosher Salt
      \item 
      \item
      \item 
      \item \textbf{Filling:}
      \item 1oz Almonds, toasted \& chopped
      \item 1 TBSP granulated sugar
      \item 2 TBSP bread crumbs
      \item $\frac{1}{2}$ cup Raspberry Jam
      \item 1 egg yolk, whisked w/ 1 tsp Water
      \item 1 tsp sparking sugar
    \end{enumerate}
  }
  \colchunk{
    \begin{enumerate}
      \item Combine all of the \textbf{DOUGH} ingredients in a food processor. pulse, process till shaggy. \textbf{1 min}
      \item Form dough on greased plastic wrap to 3x4 in rectangle. Dust and cover. 
      \item Smooth with a pin, then smooth edges.
      \item Chill to 40\textdegree F. Can store like this.
      \item
      \item
      \item Preheat oven 375\textdegree F.
      \item Sack 2 baking sheets, lined with parchment paper.
      \item In a small bowl, combine almonds and sugar.
      \item Roll Dough to 4x12in. Spread jam, almond mixture, bread crumbs.
      \item Tightly roll along 12in. Aim for 3 revolutions. Seal seam.
      \item Move to fridge on sheet for 30 min.
      \item Cut roll! Transfer to freezer for 20min. Can store like this.
      \item 
      \item
      \item Brush surface with egg wash. Sprinkle sugar. Bake 22-25min. 
      \item Cool on rack, 1 hour. 
    \end{enumerate}
  }
\end{parcolumns}



\end{recipe}
