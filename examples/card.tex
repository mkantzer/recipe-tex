\ingredient{leftover fond}
\ingredient{1 shallot, chopped}
\ingredient{olive oil}
\ingredient{1 tsp rosemary, chopped}
\ingredient{mushrooms, chopped}
\ingredient{1 TBSP jam/whatever}
\ingredient{\frac{1}{2} cup stock, or buillion cube}
\ingredient{\frac{1}{4} cup water}
\ingredient{5 tsp balsamic vinegar}
\ingredient{1 TBSP butter}
\ingredient{S\&P, to taste}
\begin{recipe}{mike's pan sauce} {5}
  Heat in pan @ medium. Add shallot, OO, rosemary, mushrooms. Cook, tossing, until softened.

  Add jam, stock, water, vinegar. Simmer 2-3 minutes, until \i{thick and saucy}

  Remove from heat. Add butter and season w/ S\&P. Stir to melt. 

  Serve over meat. 
\end{recipe}
